void winCond(){
	int max=0;
	int maxPlayer=-1;
	String ret = "";
	
	foreach player_id in roundSummary{
		int value = roundSummary[player_id].value;
		ret += "Player " + player_id + " : " + value + ", ";
		if(value>max){
			max = value;
			maxPlayer = player_id;
		}
	}
	
	Broadcast("Result:\n"+ret);
	if( maxPlayer==-1)
		Broadcast("No winner this turn");
	else
		Broadcast("Player " + maxPlayer +" win!");
	
	
}
